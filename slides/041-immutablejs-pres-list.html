Immutable List


var foo = Immutable.List([
	"lundi",
	"mardi",
	"mercredi",
	"jeudi",
	"vendredi",
	"samedi",
	"dimanche"
]);


foo = foo.push("samedi") // Ajoute l'élement "samedi" à la suite de la liste
foo = foo.filter(x => x !== "lundi") // Applique un filtre à la liste (jour !== lundi)


foo.toJS() // Array ["mardi", "mercredi", "jeudi", "vendredi", "samedi", "dimanche", "samedi"]
