var foo = Immutable.Map({
	lundi: "Mcdonald's",
	mardi: "Resto",
	mercredi: "Sandwitch",
	jeudi: "Pizza",
	vendredi: "Mcdonald's",
	samedi: null,
	dimanche: null
});


var bar = foo.get("jeudi") // Récupère la valeur "Pizza"
