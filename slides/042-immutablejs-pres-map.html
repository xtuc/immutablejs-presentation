Immutable Map


var foo = Immutable.Map({
	lundi: "Mcdonald's",
	mardi: "Gauffre (Waffle Factory)",
	mercredi: "Sandwitch",
	jeudi: "Pizza",
	vendredi: "Mcdonald's",
	samedi: null,
	dimanche: null
});


foo.get("jeudi") // Récupère la valeur "Pizza"
foo.set("samedi", "Pizza") // La valeur de la clé "samedi" est "Pizza"


foo.toJS() // Object {lundi: "Mcdonald's", mardi: "Gauffre (...)", mercredi: "Sandwitch", jeudi: "Pizza", 
// vendredi: "Mcdonald's", samedi: "Pizza", dimanche: null}
