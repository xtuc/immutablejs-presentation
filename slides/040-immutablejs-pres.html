Installation

npm i immutable
Usage

import Immutable from "Immutable";

var myList   = Immutable.List();
var myMap    = Immutable.Map();
var mySet    = Immutable.Set();
var myStack  = Immutable.Stack();
var myRecord = Immutable.Record();
var mySeq    = Immutable.Seq();


var myObj    = Immutable.fromJS(); // Transforme un objet/tableau JS en Immutable List/Map
