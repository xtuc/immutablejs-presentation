Installation

npm i immutable
Usage

import Immutable from "Immutable";

var list         = Immutable.List();
var map          = Immutable.Map();
var stack        = Immutable.Stack();
var record       = Immutable.Record();
var seq          = Immutable.Seq();


var immutalbeObj = Immutable.fromJS({ … });
var obj          = myImmutalbeObj.toJS();
list = liste, map = Tableau associatif, stack = Tableau, record = classe, seq = séquence de valeurs, fromjs = // Transforme un objet/tableau JS en Immutable List/Map, tojs = inverse
